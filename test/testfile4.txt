// test "break" and "continue" in "while stmt"

int main()
{
	int i, j, k, result;
	k = getint(); // j > 0

	// sum of k*k, k*k-1, ..., k+1, k-1, ..., 2, 1, 0
	i = k * k;
	result = 0;
	while (1)
	{
		if (i < 0) break; 
		if (i == k) { i = i - 1; continue; }
		result = result + i;
		i = i - 1;
	}
	printf("%d\n", result);

	// sum of k*k, k*k-1, ..., k+1
	i = k * k;
	result = 0;
	while (1)
	{
		if (i == k) { break; }
		result = result + i;
		i = i - 1;
	}
	printf("%d\n", result);

	// 		[1*1]	[1*2]	... [1*k]
	// sum of 	[2*1]	[2*2]	... [2*k]
	//		...	...	... ...
	//		[k*1]	[k*2]	... [k*k]
	// [n] = (n > k*k/2) ? 0 : n
	int max = k * k / 2;
	result = 0;
	i = 1;
	while (1)
	{
		if ( i > k) break;
		j = 1;	
		while (1)
		{
			if ( j > k) break;
			int tmp;
			tmp = i * j;
			if (tmp > max)
			{
				j = j + 1;
				continue;
			}
			result = result + tmp;
			j = j + 1;
		}
		i = i + 1;
	}
	printf("%d\n", result);
	return 0;
}
